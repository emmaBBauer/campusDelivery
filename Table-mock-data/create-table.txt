use campusdeliverydata;

CREATE TABLE user (
    username varchar(20),
    email varchar(30),
    password varchar(20),
    firstname varchar(20),
    lastname varchar(20),
    numberOfDeliveries int,
    class varchar(10),
    PRIMARY KEY(username,email)
);
