use campusdeliverydata;
drop table delivery;
drop table ordering;

create table user(
                    id int,
                     username varchar(20),
                     email varchar(30),
                     userPassword varchar(20),
                     firstname varchar(20),
                     lastname varchar(20),
                     numberOfDeliveries int,
                     class varchar(10),
                     primary key (id)
);

create table ordering(
        id int,
        delivery varchar(60),
        userID int,
        product varchar(15),
        quantity int,
        price double,
        currentStatus varchar(15),
        PRIMARY KEY (id),
        FOREIGN KEY (userID) REFERENCES user(id)
);

create table delivery(
    id int,
    userID int,
    shop varchar(20) NOT NULL,
    deliveryDate date,
    deliveryTime time,
    PRIMARY KEY (id),
    FOREIGN KEY (userID) REFERENCES user(id)
);
