use campusdeliverydata;
DROP TABLE IF EXISTS  ordering, delivery, user;


create table user(
    id int,
    username varchar(30),
    email varchar(40),
    userPassword varchar(30),
    firstname varchar(30),
    lastname varchar(30),
    numberOfDeliveries int,
    klasse varchar(10),
    primary key (id)
);



CREATE TABLE `delivery` (
    id int,
    userID int,
    shop varchar(30),
    deliveryDate date,
    deliveryTime time,
    PRIMARY KEY (id),
    FOREIGN KEY (userID) REFERENCES user (id)
);



create table ordering(
    id int,
    delivery int,
    userID int,
    product varchar(50),
    quantity int,
    price double,
    currentStatus varchar(15),
    notes varchar(200),
    PRIMARY KEY (id),
    FOREIGN KEY (userID) REFERENCES user(id),
    FOREIGN KEY (delivery) REFERENCES delivery(id)
);


CREATE TABLE shops (
    id int,
    name varchar(50),
    PRIMARY KEY (id)
);
