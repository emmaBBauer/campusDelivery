use campusdeliverydata;

create table user(
                     username varchar(20),
                     email varchar(30),
                     password varchar(20),
                     firstname varchar(20),
                     lastname varchar(20),
                     numberOfDeliveries int,
                     class varchar(10),
                     primary key (username, email)
);

create table ordering(
        id int,
        delivery varchar(60),
        user varchar(50),
        product varchar(15),
        quantity int,
        price double,
        status varchar(15),
        PRIMARY KEY (id),
        FOREIGN KEY (user) REFERENCES user(username)
);

create table delivery(
    user varchar(50),
    shop varchar(20) NOT NULL,
    date date,
    time time,
    PRIMARY KEY (user,date,time),
    FOREIGN KEY (user) REFERENCES user(username)
)
